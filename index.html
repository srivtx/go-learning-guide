package main

import "fmt"

func main() {

    var name string = "Go"         // Explicit type

Key Points:

  • var - Explicit variable declaration with optional type
  • := - Short declaration with type inference (inside functions only)
  • const - Compile-time constants, immutable
  • Zero values - Variables have default values (0, "", false, nil)

Functions

functions.go 
package main

import (
    "errors"
    "fmt"
)

// Basic function
func greet(name string) string {
    return "Hello, " + name + "!"
}

// Function with multiple parameters
func add(a, b int) int {
    return a + b
}

// Function with multiple return values
func divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, errors.New("division by zero")
    }
    return a / b, nil
}

// Function with named return values
func calculate(a, b int) (sum, difference, product int) {
    sum = a + b
    difference = a - b
    product = a * b
    return // naked return
}

// Variadic function (accepts variable number of arguments)
func sum(numbers ...int) int {
    total := 0
    for _, num := range numbers {
        total += num
    }
    return total
}

// Function as a type
type Calculator func(int, int) int

func multiply(a, b int) int {
    return a * b
}

// Higher-order function
func operate(a, b int, fn Calculator) int {
    return fn(a, b)
}

// Anonymous function (closure)
func getMultiplier(factor int) func(int) int {
    return func(n int) int {
        return n * factor
    }
}

// Method with receiver
type Rectangle struct {
    width, height float64
}

func (r Rectangle) Area() float64 {
    return r.width * r.height
}

func (r *Rectangle) Scale(factor float64) {
    r.width *= factor
    r.height *= factor
}

func main() {
    // Basic function calls
    fmt.Println(greet("Gopher"))
    fmt.Println("Sum:", add(5, 3))
    
    // Multiple return values
    result, err := divide(10, 2)
    if err != nil {
        fmt.Println("Error:", err)
    } else {
        fmt.Printf("Division result: %.2f\n", result)
    }
    
    // Named returns
    s, d, p := calculate(10, 3)
    fmt.Printf("Sum: %d, Diff: %d, Product: %d\n", s, d, p)
    
    // Variadic function
    fmt.Println("Sum of multiple numbers:", sum(1, 2, 3, 4, 5))
    
    // Function as variable
    var calc Calculator = multiply
    fmt.Println("Multiply function:", operate(4, 5, calc))
    
    // Anonymous function
    doubler := getMultiplier(2)
    fmt.Println("Double 7:", doubler(7))
    
    // Methods
    rect := Rectangle{width: 5, height: 3}
    fmt.Printf("Rectangle area: %.2f\n", rect.Area())
    rect.Scale(2)
    fmt.Printf("Scaled rectangle area: %.2f\n", rect.Area())
}

Function Features:

  • Multiple returns: Functions can return multiple values
  • Named returns: Return values can be named
  • Variadic functions: Accept variable number of arguments
  • First-class functions: Functions are values
  • Closures: Functions can capture variables from outer scope
  • Methods: Functions with receivers (like methods in OOP)

Interfaces and Polymorphism

interfaces.go 
package main

import (
    "fmt"
    "math"
)

// Interface definition
type Shape interface {
    Area() float64
    Perimeter() float64
}

// Interface composition
type Drawable interface {
    Draw()
}

type DrawableShape interface {
    Shape
    Drawable
}

// Rectangle struct implementing Shape
type Rectangle struct {
    width, height float64
}

func (r Rectangle) Area() float64 {
    return r.width * r.height
}

func (r Rectangle) Perimeter() float64 {
    return 2 * (r.width + r.height)
}

func (r Rectangle) Draw() {
    fmt.Printf("Drawing rectangle: %.1f x %.1f\n", r.width, r.height)
}

// Circle struct implementing Shape
type Circle struct {
    radius float64
}

func (c Circle) Area() float64 {
    return math.Pi * c.radius * c.radius
}

func (c Circle) Perimeter() float64 {
    return 2 * math.Pi * c.radius
}

func (c Circle) Draw() {
    fmt.Printf("Drawing circle with radius: %.1f\n", c.radius)
}

// Function that works with any Shape
func printShapeInfo(s Shape) {
    fmt.Printf("Area: %.2f, Perimeter: %.2f\n", s.Area(), s.Perimeter())
}

// Type assertion and type switch
func describe(i interface{}) {
    switch v := i.(type) {
    case Rectangle:
        fmt.Printf("Rectangle with width %.1f and height %.1f\n", v.width, v.height)
    case Circle:
        fmt.Printf("Circle with radius %.1f\n", v.radius)
    case string:
        fmt.Printf("String: %s\n", v)
    case int:
        fmt.Printf("Integer: %d\n", v)
    default:
        fmt.Printf("Unknown type: %T\n", v)
    }
}

// Empty interface example
func printAnything(value interface{}) {
    fmt.Printf("Value: %v, Type: %T\n", value, value)
}

func main() {
    // Creating shapes
    rect := Rectangle{width: 5, height: 3}
    circle := Circle{radius: 4}
    
    // Using interface polymorphism
    shapes := []Shape{rect, circle}
    
    fmt.Println("Shape information:")
    for i, shape := range shapes {
        fmt.Printf("Shape %d:\n", i+1)
        printShapeInfo(shape)
        
        // Type assertion to check if shape implements Drawable
        if drawable, ok := shape.(Drawable); ok {
            drawable.Draw()
        }
        fmt.Println()
    }
    
    // Type assertion examples
    var s Shape = rect
    if r, ok := s.(Rectangle); ok {
        fmt.Printf("Successfully asserted Rectangle: %+v\n", r)
    }
    
    // Type switch examples
    fmt.Println("Type switch examples:")
    describe(rect)
    describe(circle)
    describe("Hello, Go!")
    describe(42)
    describe(3.14)
    
    // Empty interface examples
    fmt.Println("Empty interface examples:")
    printAnything(42)
    printAnything("Hello")
    printAnything(rect)
    printAnything([]int{1, 2, 3})
}

Interface Concepts:

  • Implicit implementation: No "implements" keyword needed
  • Interface composition: Embed interfaces in other interfaces
  • Type assertion: Extract concrete type from interface
  • Type switch: Handle different types in one construct
  • Empty interface: interface{} can hold any value

Concurrency: Goroutines and Channels

concurrency.go 
package main

import (
    "fmt"
    "sync"
    "time"
)

// Basic goroutine example
func sayHello(name string, wg *sync.WaitGroup) {
    defer wg.Done()
    for i := 0; i < 3; i++ {
        fmt.Printf("Hello from %s - %d\n", name, i)
        time.Sleep(100 * time.Millisecond)
    }
}

// Channel communication
func producer(ch chan<- int, wg *sync.WaitGroup) {
    defer wg.Done()
    defer close(ch)
    
    for i := 1; i <= 5; i++ {
        ch <- i
        fmt.Printf("Produced: %d\n", i)
        time.Sleep(200 * time.Millisecond)
    }
}

func consumer(ch <-chan int, wg *sync.WaitGroup) {
    defer wg.Done()
    
    for value := range ch {
        fmt.Printf("Consumed: %d\n", value)
        time.Sleep(150 * time.Millisecond)
    }
}

// Worker pool pattern
func worker(id int, jobs <-chan int, results chan<- int, wg *sync.WaitGroup) {
    defer wg.Done()
    
    for job := range jobs {
        fmt.Printf("Worker %d processing job %d\n", id, job)
        time.Sleep(time.Millisecond * 100)
        results <- job * 2
    }
}

// Select statement for non-blocking operations
func selectExample() {
    ch1 := make(chan string)
    ch2 := make(chan string)
    
    go func() {
        time.Sleep(200 * time.Millisecond)
        ch1 <- "from channel 1"
    }()
    
    go func() {
        time.Sleep(100 * time.Millisecond)
        ch2 <- "from channel 2"
    }()
    
    for i := 0; i < 2; i++ {
        select {
        case msg1 := <-ch1:
            fmt.Println("Received:", msg1)
        case msg2 := <-ch2:
            fmt.Println("Received:", msg2)
        case <-time.After(300 * time.Millisecond):
            fmt.Println("Timeout!")
            return
        }
    }
}

// Buffered channels
func bufferedChannelExample() {
    ch := make(chan int, 3) // Buffered channel with capacity 3
    
    // Send without blocking (until buffer is full)
    ch <- 1
    ch <- 2
    ch <- 3
    
    fmt.Printf("Channel length: %d, capacity: %d\n", len(ch), cap(ch))
    
    // Receive values
    for i := 0; i < 3; i++ {
        value := <-ch
        fmt.Printf("Received: %d\n", value)
    }
}

// Mutex for shared data protection
type Counter struct {
    mu    sync.Mutex
    value int
}

func (c *Counter) Increment() {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.value++
}

func (c *Counter) Value() int {
    c.mu.Lock()
    defer c.mu.Unlock()
    return c.value
}

func main() {
    fmt.Println("=== Basic Goroutines ===")
    var wg sync.WaitGroup
    
    wg.Add(2)
    go sayHello("Goroutine 1", &wg)
    go sayHello("Goroutine 2", &wg)
    wg.Wait()
    
    fmt.Println("\n=== Producer-Consumer Pattern ===")
    ch := make(chan int)
    wg.Add(2)
    
    go producer(ch, &wg)
    go consumer(ch, &wg)
    wg.Wait()
    
    fmt.Println("\n=== Worker Pool Pattern ===")
    jobs := make(chan int, 5)
    results := make(chan int, 5)
    
    // Start workers
    wg.Add(3)
    for w := 1; w <= 3; w++ {
        go worker(w, jobs, results, &wg)
    }
    
    // Send jobs
    for j := 1; j <= 5; j++ {
        jobs <- j
    }
    close(jobs)
    
    // Wait for workers to finish
    go func() {
        wg.Wait()
        close(results)
    }()
    
    // Collect results
    for result := range results {
        fmt.Printf("Result: %d\n", result)
    }
    
    fmt.Println("\n=== Select Statement ===")
    selectExample()
    
    fmt.Println("\n=== Buffered Channels ===")
    bufferedChannelExample()
    
    fmt.Println("\n=== Mutex Example ===")
    counter := &Counter{}
    
    // Start multiple goroutines to increment counter
    wg.Add(10)
    for i := 0; i < 10; i++ {
        go func() {
            defer wg.Done()
            for j := 0; j < 100; j++ {
                counter.Increment()
            }
        }()
    }
    
    wg.Wait()
    fmt.Printf("Final counter value: %d\n", counter.Value())
}

Concurrency Concepts:

  • Goroutines: Lightweight threads managed by Go runtime
  • Channels: Communication mechanism between goroutines
  • Select: Choose from multiple channel operations
  • Buffered channels: Allow non-blocking sends up to buffer size
  • WaitGroup: Wait for multiple goroutines to complete
  • Mutex: Protect shared data from race conditions

Error Handling and Panic Recovery

error_handling.go 
package main

import (
    "errors"
    "fmt"
    "log"
    "os"
    "strconv"
)

// Custom error type
type ValidationError struct {
    Field   string
    Message string
}

func (e ValidationError) Error() string {
    return fmt.Sprintf("validation error in field '%s': %s", e.Field, e.Message)
}

// Multiple error handling patterns
func divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, errors.New("division by zero")
    }
    return a / b, nil
}

// Wrapped errors
func processFile(filename string) error {
    if filename == "" {
        return fmt.Errorf("processFile: %w", errors.New("filename cannot be empty"))
    }
    
    // Simulate file operation that might fail
    if filename == "invalid.txt" {
        return fmt.Errorf("processFile: failed to open %s: %w", filename, os.ErrNotExist)
    }
    
    return nil
}

// Custom error with additional context
func validateUser(name string, age int) error {
    if name == "" {
        return ValidationError{
            Field:   "name",
            Message: "name cannot be empty",
        }
    }
    
    if age < 0 || age > 150 {
        return ValidationError{
            Field:   "age",
            Message: "age must be between 0 and 150",
        }
    }
    
    return nil
}

// Panic and recover example
func riskyOperation(value int) (result int) {
    defer func() {
        if r := recover(); r != nil {
            fmt.Printf("Recovered from panic: %v\n", r)
            result = -1 // Set a safe default value
        }
    }()
    
    if value == 0 {
        panic("cannot process zero value")
    }
    
    return 100 / value
}

// Multiple return pattern for errors
func parseAndValidate(input string) (int, error) {
    // First, try to parse
    num, err := strconv.Atoi(input)
    if err != nil {
        return 0, fmt.Errorf("failed to parse '%s': %w", input, err)
    }
    
    // Then validate
    if num < 0 {
        return 0, errors.New("number must be non-negative")
    }
    
    return num, nil
}

// Error checking with early return
func processNumbers(inputs []string) error {
    for i, input := range inputs {
        num, err := parseAndValidate(input)
        if err != nil {
            return fmt.Errorf("error processing input %d ('%s'): %w", i, input, err)
        }
        fmt.Printf("Processed: %d\n", num)
    }
    return nil
}

// Sentinel errors
var (
    ErrNotFound     = errors.New("item not found")
    ErrInvalidInput = errors.New("invalid input")
    ErrTimeout      = errors.New("operation timed out")
)

func findItem(id int) (string, error) {
    items := map[int]string{
        1: "apple",
        2: "banana",
        3: "orange",
    }
    
    if item, exists := items[id]; exists {
        return item, nil
    }
    
    return "", ErrNotFound
}

func main() {
    fmt.Println("=== Basic Error Handling ===")
    result, err := divide(10, 2)
    if err != nil {
        fmt.Printf("Error: %v\n", err)
    } else {
        fmt.Printf("Result: %.2f\n", result)
    }
    
    // Error case
    _, err = divide(10, 0)
    if err != nil {
        fmt.Printf("Error: %v\n", err)
    }
    
    fmt.Println("\n=== Wrapped Errors ===")
    err = processFile("")
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        
        // Unwrap errors
        if unwrapped := errors.Unwrap(err); unwrapped != nil {
            fmt.Printf("Unwrapped: %v\n", unwrapped)
        }
    }
    
    fmt.Println("\n=== Custom Errors ===")
    err = validateUser("", 25)
    if err != nil {
        fmt.Printf("Validation failed: %v\n", err)
        
        // Type assertion for custom error
        if validationErr, ok := err.(ValidationError); ok {
            fmt.Printf("Field: %s, Message: %s\n", validationErr.Field, validationErr.Message)
        }
    }
    
    fmt.Println("\n=== Panic and Recover ===")
    fmt.Printf("Result 1: %d\n", riskyOperation(10))
    fmt.Printf("Result 2: %d\n", riskyOperation(0)) // This will panic and recover
    fmt.Printf("Result 3: %d\n", riskyOperation(5))
    
    fmt.Println("\n=== Error Propagation ===")
    inputs := []string{"10", "20", "invalid", "30"}
    err = processNumbers(inputs)
    if err != nil {
        fmt.Printf("Processing failed: %v\n", err)
    }
    
    fmt.Println("\n=== Sentinel Errors ===")
    item, err := findItem(1)
    if err != nil {
        if errors.Is(err, ErrNotFound) {
            fmt.Println("Item not found - handling gracefully")
        } else {
            fmt.Printf("Unexpected error: %v\n", err)
        }
    } else {
        fmt.Printf("Found item: %s\n", item)
    }
    
    // Not found case
    _, err = findItem(999)
    if errors.Is(err, ErrNotFound) {
        fmt.Println("Item 999 not found - as expected")
    }
    
    fmt.Println("\n=== Logging Errors ===")
    log.SetFlags(log.LstdFlags | log.Lshortfile)
    
    if err := processFile("invalid.txt"); err != nil {
        log.Printf("File processing failed: %v", err)
    }
}

Error Handling Best Practices:

  • Explicit error checking: Always check and handle errors
  • Error wrapping: Add context using fmt.Errorf with %w verb
  • Custom errors: Implement error interface for rich error information
  • Sentinel errors: Predefined error values for comparison
  • Panic/recover: Use sparingly, only for truly exceptional conditions
  • Early returns: Check errors immediately and return early

Goroutines (Concurrency)

‚ñº
goroutines.go 
package main

import (
    "fmt"
    "time"
)

func worker(id int, jobs <-chan int, results chan<- int) {
    for j := range jobs {
        fmt.Printf("Worker %d processing job %d\n", id, j)
        time.Sleep(time.Millisecond * 100)
        results <- j * 2
    }
}

func main() {
    jobs := make(chan int, 100)
    results := make(chan int, 100)
    
    // Start 3 workers
    for w := 1; w <= 3; w++ {
        go worker(w, jobs, results)
    }
    
    // Send 5 jobs
    for j := 1; j <= 5; j++ {
        jobs <- j
    }
    close(jobs)
    
    // Collect results
    for r := 1; r <= 5; r++ {
        <-results
    }
    
    fmt.Println("All jobs completed")
}

Goroutines are lightweight threads. Channels provide safe communication between goroutines.

package main

import (
    "fmt"
    "os"
    "strconv"
)

func main() {
    if len(os.Args) != 4 {
        fmt.Println("Usage: calculator   ")
        fmt.Println("Operations: +, -, *, /")
        os.Exit(1)
    }
    
    num1, err := strconv.ParseFloat(os.Args[1], 64)
    if err != nil {
        fmt.Printf("Invalid first number: %s\n", os.Args[1])
        os.Exit(1)
    }
    
    operation := os.Args[2]
    
    num2, err := strconv.ParseFloat(os.Args[3], 64)
    if err != nil {
        fmt.Printf("Invalid second number: %s\n", os.Args[3])
        os.Exit(1)
    }
    
    result, err := calculate(num1, operation, num2)
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        os.Exit(1)
    }
    
    fmt.Printf("%.2f %s %.2f = %.2f\n", num1, operation, num2, result)
}

func calculate(a float64, op string, b float64) (float64, error) {
    switch op {
    case "+":
        return a + b, nil
    case "-":
        return a - b, nil
    case "*":
        return a * b, nil
    case "/":
        if b == 0 {
            return 0, fmt.Errorf("division by zero")
        }
        return a / b, nil
    default:
        return 0, fmt.Errorf("unsupported operation: %s", op)
    }
}

    Completed   bool      `json:"completed"`
    CreatedAt   time.Time `json:"created_at"`
}

    Items  []TodoItem `json:"items"`
    NextID int        `json:"next_id"`
}

func (tl *TodoList) AddItem(title, description string) {
    item := TodoItem{
        ID:          tl.NextID,
        Title:       title,
        Description: description,
        Completed:   false,
        CreatedAt:   time.Now(),
    }
    tl.Items = append(tl.Items, item)
    tl.NextID++
}

func (tl *TodoList) CompleteItem(id int) error {
    for i := range tl.Items {
        if tl.Items[i].ID == id {
            tl.Items[i].Completed = true
            return nil
        }
    }
    return fmt.Errorf("item with ID %d not found", id)
}

func (tl *TodoList) RemoveItem(id int) error {
    for i, item := range tl.Items {
        if item.ID == id {
            tl.Items = append(tl.Items[:i], tl.Items[i+1:]...)
            return nil
        }
    }
    return fmt.Errorf("item with ID %d not found", id)
}

func (tl *TodoList) ListItems() {
    if len(tl.Items) == 0 {
        fmt.Println("No todo items found.")
        return
    }
    
    for _, item := range tl.Items {
        status := "[ ]"
        if item.Completed {
            status = "[‚úì]"
        }
        fmt.Printf("%s %d: %s - %s\n", status, item.ID, item.Title, item.Description)
    }
}
